actor Dragunov : Weapon
{
    +WEAPON.NOAUTOFIRE
    +AMMO_OPTIONAL
    Inventory.PickupMessage "You got the Dragunov, a semi automatic sniper rifle"
    Weapon.AmmoGive1 30
    Weapon.AmmoGive2 30
    Weapon.AmmoType1 "Drnvclip"
    Weapon.AmmoType2 "SniperAmmo"
    Weapon.AmmoUse1 1
    Weapon.SelectionOrder 400
    Obituary "-"
    Scale 0.7
    AttackSound "weapons/dgvfire"
    States
    {
      Ready:
        TKGN D 1 A_WeaponReady
        Loop
      Select:
        TKGN D 1 A_Raise
        loop
      Deselect:
        TKGN D 1 A_Lower
        loop
      Fire:
        TKGN D 0 A_JumpIfInventory("Drnvclip", 1, 2)
        TKGN D 0
        goto AltFire
        TKGN D 0
        TKGN C 2 bright A_FireBullets(1, 1, 1, 60, "Bulletpuff", 1)
        TKGN B 2 bright
        TKGN D 5
        goto Ready
      AltFire:
        TKGN A 0 A_JumpIfInventory("Drnvclip", 15, 2)
        TKGN A 0 A_JumpIfInventory("SniperAmmo", 1, 2)
        TKGN A 0
        goto Ready
        TKGN E 0
        TKGN E 3
        TKGN F 3
        TKGN G 3
        TKGN H 3
        TKGN I 3
        TKGN K 0 A_PlaySound("weapons/dgvload", CHAN_WEAPON)
        TKGN K 3
        TKGN L 3
        TKGN M 3
        TKGN L 3
        TKGN M 3
        TKGN N 3
        TKGN O 3
        TKGN P 3
        TKGN Q 3
        TKGN R 3
        TKGN D 3
        TKGN D 0
        TKGN D 0 A_TakeInventory("SniperAmmo", 1)
        TKGN D 0 A_GiveInventory("Drnvclip", 1)
        TKGN D 0 A_JumpIfInventory("Drnvclip", 15, 4)
        TKGN D 0 A_JumpIfInventory("SniperAmmo", 1, 2)
        TKGN D 0
        goto AltFire+28
        TKGN D 0
        goto AltFire+22
        TKGN D 1
        goto Ready
      Spawn:
        TKGZ A -1
        stop
    }
}
