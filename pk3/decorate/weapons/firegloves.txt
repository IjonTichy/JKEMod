actor Firegloves : Weapon
{
    +WEAPON.NOAUTOAIM
    Weapon.SelectionOrder 560
    Weapon.AmmoUse1 30
    Weapon.AmmoUse2 50
    Weapon.AmmoGive1 600
    Weapon.AmmoGive2 200
    Weapon.AmmoType1 "FireAmmo"
    Weapon.AmmoType2 "JKEMana"
    Weapon.YAdjust 10
    Inventory.PickupMessage "$JKE_PK_FIREGLOVES"
    Obituary "$JKE_MP_FIREGLOVES"
    States
    {
      Deselect:
        JFLM AAABBBCCC 1 bright A_Lower
        loop
      Select:
        JFLM AAABBBCCC 1 bright A_Raise
        loop

      Ready:
        JFLM ABC 3 bright A_WeaponReady
        loop

      Fire:
        JFLM D 14 bright A_PlaySound("jkemod/firegloves/fire", CHAN_WEAPON)
        JFLM D 0 A_PlaySound("jkemod/firegloves/crackle", CHAN_WEAPON)
        JFLM E 1 bright bright A_FireBullets(0, 0, 1, 10, "RfireA", 1)
        JFLM E 1 bright A_Light1
        JFLM F 2 bright A_FireBullets(0, 0, 1, 12, "RfireB", 1)
        JFLM E 2 bright A_FireBullets(0, 0, 1, 12, "RfireA", 0)
        JFLM F 2 bright A_FireBullets(0, 0, 1, 12, "RfireB", 0)
        JFLM E 2 bright A_FireBullets(0, 0, 1, 12, "RfireC", 0)
        JFLM F 2 bright A_FireBullets(0, 0, 1, 12, "RfireD", 0)
        JFLM E 2 bright A_FireBullets(0, 0, 1, 12, "RfireE", 0)
        JFLM F 2 bright A_FireBullets(0, 0, 1, 12, "RfireD", 0)
        JFLM E 2 bright A_FireBullets(0, 0, 1, 12, "RfireC", 0)
        JFLM F 2 bright A_FireBullets(0, 0, 1, 12, "RfireD", 0)
        JFLM E 2 bright A_FireBullets(0, 0, 1, 12, "RfireE", 0)
        JFLM F 2 bright A_FireBullets(0, 0, 1, 12, "RfireD", 0)
        JFLM E 2 bright A_FireBullets(0, 0, 1, 12, "RfireC", 0)
        JFLM F 2 bright A_FireBullets(0, 0, 1, 12, "RfireD", 0)
        JFLM E 2 bright A_FireBullets(0, 0, 1, 12, "RfireE", 0)
        JFLM F 2 bright A_FireBullets(0, 0, 1, 12, "RfireD", 0)
        JFLM E 2 bright A_FireBullets(0, 0, 1, 12, "RfireC", 0)

        JFLM E 0 A_PlaySound("jkemod/firegloves/crackle", CHAN_WEAPON)
        JFLM F 2 bright A_FireBullets(0, 0, 1, 12, "RfireD", 0)
        JFLM E 2 bright A_FireBullets(0, 0, 1, 12, "RfireE", 0)
        JFLM F 2 bright A_FireBullets(0, 0, 1, 12, "RfireD", 0)
        JFLM E 2 bright A_FireBullets(0, 0, 1, 12, "RfireC", 0)
        JFLM F 2 bright A_FireBullets(0, 0, 1, 12, "RfireD", 0)
        JFLM E 2 bright A_FireBullets(0, 0, 1, 12, "RfireE", 0)

        JFLM F 2 bright A_PlaySound("jkemod/firegloves/explode", CHAN_WEAPON)
        JFLM E 2 bright A_FireBullets(0, 0, 1, 0, "GloveFireBig", 0)
        JFLM FEFEFEFEFEFE 2 bright
        JFLM G 15 bright A_Light0
        Goto Ready


      AltFire:
        JFLM D 20 bright
        JFLM E 2 bright A_Light1
        JFLM F 2 bright
        JFLM E 2 bright A_FireCustomMissile("FireGloveFireballSmall", 0, 1, 0, 0)
        JFLM FE 2 bright
        JFLM F 2 bright A_FireCustomMissile("FireGloveFireballSmall", 0, 0, 0, 0)
        JFLM EF 2 bright
        JFLM E 2 bright A_FireCustomMissile("FireGloveFireball", 3, 0, 0, 0)
        JFLM F 2 bright
        JFLM E 2 bright A_FireCustomMissile("FireGloveFireball", -3, 0, 0, 0)
        JFLM FE 2 bright
        JFLM F 2 bright A_FireCustomMissile("FireGloveFireballSmall", 0, 0, 0, 0)
        JFLM EF 2 bright
        JFLM E 2 bright A_FireCustomMissile("FireGloveMeteor", 0, 0, 0, 0)
        JFLM FE 2 bright
        JFLM F 2 bright A_FireCustomMissile("FireGloveMeteor", 0, 0, 0, 0)
        JFLM E 2 bright
        JFLM F 2 bright A_FireCustomMissile("FireGloveFireballSmall", 0, 0, 0, 0)
        JFLM EF 2 bright
        JFLM E 2 bright A_FireCustomMissile("FireGloveGroundFireball", 4, 0, 0, 0)
        JFLM F 2 bright A_FireCustomMissile("FireGloveGroundFireball", -4, 0, 0, 0)
        JFLM E 2 bright
        JFLM F 2 bright A_FireCustomMissile("FireGloveFireballSmall", 0, 0, 0, 0)
        JFLM G 20 bright A_Light0
        Goto Ready
      Spawn:
        JFLM Z -1 bright
        stop
    }
}
