Actor Ballesta: Weapon
{
    Weapon.SelectionOrder 560
    Weapon.AmmoUse 1
    Weapon.AmmoGive 30
    Weapon.AmmoType "BallestaAmmo"
    Weapon.upsound "weapons/load2"
    Inventory.PickupMessage "$JKE_PK_BALLESTA"
    Obituary "$JKE_MP_BALLESTA"
    States
    {
      Ready:
        XBOW H 1 A_WeaponReady
        Loop
      Deselect:
        XBOW H 1 A_Lower
        Loop
      Select:
        XBOW H 1 A_Raise
        Loop
      Fire:
        XBOW B 0 A_PlaySound("jkemod/ballesta/fire", CHAN_WEAPON)
        XBOW B 4 bright A_FireCustomMissile("BallestaArrow", 0, 1)
        XBOW F 4
        XBOW J 3
        XBOW I 3 A_ReFire
        Goto Ready
      Spawn:
        CBOW A -1
        Stop
    }
}


actor BallestaArrow
{
    Radius 5
    Height 5
    Gravity 0.25

    Projectile
    +EXPLODEONWATER

    Damage 20
    Speed 70

    DeathSound "jkemod/ballesta/explode"

    States
    {
      Spawn:
        ARRW A -1
        stop

      Death:
        CFCF Q 0 A_Explode(15, 30, 0)
        CFCF QRSTUVWXYZ 2
        stop
    }
}
