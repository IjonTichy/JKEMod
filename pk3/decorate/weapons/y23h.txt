
actor " Y23H " : Weapon
{
    Weapon.AmmoType1 "Cell"
    Weapon.AmmoUse1 3
    Weapon.AmmoGive1 100

    Weapon.SelectionOrder 600

    Inventory.PickupMessage "$JKE_PK_Y23H"

    States
    {
      Ready:
        Y23H A 1 A_WeaponReady
        Loop
      Select:
        Y23H A 1 A_Raise
        loop
      Deselect:
        Y23H A 1 A_Lower
        loop
      Fire:
        Y23H A 0 A_FireCustomMissile("Y23HPlasma",       0,  1, 0, 0)
        Y23H A 0 A_FireCustomMissile("Y23HPlasmaSmall",  12, 0, 0, 0)
        Y23H A 0 A_FireCustomMissile("Y23HPlasmaSmall", -12, 0, 0, 0)
        Y23H D 2 bright
        Y23H C 1 bright
        Y23H B 12 A_ReFire
        goto Ready
      Spawn:
        Y23H Z -1
        stop
    }
}

actor Y23HPlasma
{
    Radius 10
    Height 10
    Speed 30

    Projectile
    Damage 0

    +RANDOMIZE
    +FORCERADIUSDMG
    +NODAMAGETHRUST

    RenderStyle Add
    Scale 1.3

    SeeSound   "jkemod/y23h/fire"
    DeathSound "jkemod/y23h/explode"

    Obituary "$JKE_MP_Y23H"

    States
    {
      Spawn:
        RPLS AAABBB 2 bright A_SpawnItemEx("Y23HTrail")
        loop

      Death:
        RPLS A 0 A_Explode(48, 96, 0)
        RPLS CDEFG 4 bright
        stop
    }
}
