

actor IAmTheBetterPlayer: Inventory { +UNDROPPABLE inventory.maxamount 1 }

actor BetterAmmoGiver: CustomInventory
{
    +AUTOACTIVATE
    States
    {
      Spawn:
        TNT1 A 0
        stop

      Pickup:
      Use:
        TNT1 A 0 A_GiveInventory("Backpack", 1)
        TNT1 A 0 A_GiveInventory("Clip",        10000)
        TNT1 A 0 A_GiveInventory("Shell",       10000)
        TNT1 A 0 A_GiveInventory("RocketAmmo",  10000)
        TNT1 A 0 A_GiveInventory("Cell",        10000)
        stop
    }
}



actor BetterPlayer: DoomPlayer
{
    Player.DisplayName "Doomguy"
    Player.StartItem "IAmTheBetterPlayer"
    Player.StartItem "BetterAmmoGiver"

    Player.StartItem " Pistol "
    Player.StartItem " Fists "
    Player.StartItem " Shotgun "
    Player.StartItem "Super Shotgun"
    Player.StartItem " Chaingun "
    Player.StartItem "Rocket Launcher"
    Player.StartItem "Plasma Rifle"
    Player.CrouchSprite "BPLC"

    States
    {
      Spawn:
        BPLY A -1
        Loop
      See:
        BPLY ABCD 4
        Loop
      Missile:
        BPLY E 12
        Goto Spawn
      Melee:
        BPLY F 6 BRIGHT
        Goto Missile
      Pain:
        BPLY G 4
        BPLY G 4 A_Pain
        Goto Spawn
      Death:
        BPLY H 0 A_PlayerSkinCheck("AltSkinDeath")
      Death1:
        BPLY H 10
        BPLY I 10 A_PlayerScream
        BPLY J 10 A_NoBlocking
        BPLY KLM 10
        BPLY N -1
        Stop
      XDeath:
        BPLY O 0 A_PlayerSkinCheck("AltSkinXDeath")
      XDeath1:
        BPLY O 5
        BPLY P 5 A_XScream
        BPLY Q 5 A_NoBlocking
        BPLY RSTUV 5
        BPLY W -1
        Stop
      AltSkinDeath:
        BPLY H 6
        BPLY I 6 A_PlayerScream
        BPLY JK 6
        BPLY L 6 A_NoBlocking
        BPLY MNO 6
        BPLY P -1
        Stop
      AltSkinXDeath:
        BPLY Q 5 A_PlayerScream
        BPLY R 0 A_NoBlocking
        BPLY R 5 A_SkullPop
        BPLY STUVWX 5
        BPLY Y -1
        Stop
    }

}
