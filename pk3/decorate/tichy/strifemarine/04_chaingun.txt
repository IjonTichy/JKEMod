actor " Power S-Chaingun ": Weapon
{
    Weapon.AmmoType1 "Clip"
    Weapon.AmmoType2 "RocketAmmo"
    Weapon.AmmoUse1 1
    Weapon.AmmoUse2 1

    AttackSound "hexenmarine/chaingunfire"
    Decal "BulletChip"

    Weapon.SisterWeapon " S-Chaingun "
    +POwERED_UP

    Obituary "$STRIFEMARINE_MP_DWEAPCHAINGUN_POWER"

    States
    {
      Spawn:
        HXCG Z -1
        Loop
      Ready:
        HXCG A 1 A_WeaponReady
        Loop
      Deselect:
        HXCG A 0 A_StopSound(CHAN_BODY)
        HXCG A 0 A_Lower
        HXCG A 1 A_Lower
        Goto Deselect
      Select:
        HXCG A 0 A_Raise
        HXCG A 1 A_Raise
        Goto Select

      Fire:
        HXCG F 0 A_JumpIfInventory("Clip", 4, 1)
        goto Empty
        HXCG F 1 bright A_FireBullets(1, 1, -1, 6, "ExplosiveSChaingunBulletPuff")
        HXCG B 2
        HXCG F 0 A_JumpIfInventory("Clip", 4, 1)
        goto Empty
        HXCG G 1 bright A_FireBullets(1, 1, -1, 6, "ExplosiveSChaingunBulletPuff")
        HXCG DEE 1
        HXCG BCDDEEAABBBCCCDDDDEEEEE 1 A_Refire
        goto Ready

      AltFire:
        HXCG F 0 A_JumpIfInventory("RocketAmmo", 1, 1)
        goto Empty
        HXCG F 0 A_PlaySound("hexenmarine/chaingunaltfire", CHAN_WEAPON)
        HXCG F 1 bright A_FireCustomMissile("SChaingunGrenade", random(-4, 4)/2.0, 1, 6, 0, random(-4, 4)/2.0)
        HXCG B 2
        HXCG CDE 1
        HXCG BCDDEEAABBBCCCDDDDEEEEE 1 A_Refire
        goto Ready

      Empty:
        HXCG F 0 A_StopSound(CHAN_BODY)
        HXCG A 2
        HXCG B 10
        HXCG A 2 A_PlaySound("hexenmarine/noammo")
        goto Ready
    }
}
