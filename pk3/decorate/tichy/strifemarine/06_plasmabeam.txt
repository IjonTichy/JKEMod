
actor " S-Plasma Beam ": Weapon
{
    Weapon.SelectionOrder 500
    Weapon.SlotNumber 6

    Weapon.AmmoType1 "Cell"
    Weapon.AmmoType2 "Cell"
    Weapon.AmmoUse1 1
    Weapon.AmmoUse2 1
    Weapon.AmmoGive1 75

    Inventory.PickupMessage "$STRIFEMARINE_PK_DWEAPPLASMABEAM"
    Obituary "$STRIFEMARINE_MP_DWEAPPLASMABEAM"

    states
    {
      Ready:
        PLSB A 1 A_WeaponReady
        loop
      Deselect:
        PLSB A 1 A_Lower
        PLSB A 0 A_StopSoundEx (soundslot7)
        loop
      Select:
        PLSB A 1 A_Raise
        PLSB A 0 A_StopSoundEx (soundslot7)
        loop
      Fire:
        PLSB A 0 A_GunFlash
        PLSB A 0 A_PlaySound("hexenmarine/beamproj", CHAN_WEAPON)
        PLSB A 4 bright A_FireCustomMIssile("BeamShot", 0, 1, random(-3,3), random(-3,3))
        PLSB A 0 A_ReFire
        PLSB A 0 A_PlaySound("hexenmarine/beamprojstop", CHAN_WEAPON)
        goto Ready
      AltFire:
        PLSB A 0 bright A_GunFlash
        PLSB A 0 A_PlaySound("hexenmarine/beamfire", CHAN_WEAPON)
        PLSB A 1 bright A_FireCustomMissile("PlasmaBeamProj", 0, 1)
        PLSB A 0 A_PlaySound("hexenmarine/beamfire", CHAN_WEAPON)
        PLSB A 1 bright A_FireCustomMissile("PlasmaBeamProj", 0, 0)
        PLSB A 0 A_PlaySound("hexenmarine/beamfire", CHAN_WEAPON)
        PLSB A 1 bright A_FireCustomMissile("PlasmaBeamProj", 0, 0)
        PLSB A 0 A_PlaySound("hexenmarine/beamfire", CHAN_WEAPON)
        PLSB A 1 bright A_FireCustomMissile("PlasmaBeamProj", 0, 0)
        PLSB A 0 A_PlaySound("hexenmarine/beamfire", CHAN_WEAPON)
        PLSB A 1 bright A_FireCustomMissile("PlasmaBeamProj", 0, 0)
        PLSB A 1 A_ReFire
        PLSB A 0 A_PlaySound("hexenmarine/beamprojstop", CHAN_WEAPON)
        goto Ready
      Flash:
        PLSB B 2 bright A_Light1
        PLSB C 3 bright A_Light2
        PLSB A 0 bright A_Light0
        stop
      Spawn:
        PBPK A -1
        Stop
    }
}

{
    Height 1
    Radius 0.2

    Speed 512
    Damage (5)
    +PUFFONACTORS
    +ALWAYSPUFF

    RenderStyle Add
    MissileType PlasmaBeamTrail

    Decal DoomImpScorch
    Scale 0.25

    States
    {
      Spawn:
        PLSP A 1
        // fall-through

      Death:
        PLSP A 0 A_Stop
        PLSP ABCDE 4 bright
        stop
    }
}
